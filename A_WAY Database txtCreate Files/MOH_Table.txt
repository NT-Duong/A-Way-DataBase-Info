CREATE TABLE MOH(
MOH_ID INT NOT NULL AUTO_INCREMENT, 
MOH_Name VARCHAR(255) NOT NULL,
MOH_Type VARCHAR(45) NOT NULL,
MOH_Price Decimal(9,2) NOT NULL,
MOH_Quantity INT NOT NULL,
MOH_Purchased DATE NOT NULL, 
Contract_ID INT NOT NULL, 
PRIMARY KEY(TOH_ID),
Foreign KEY(Contract_ID) REFERENCE Contracts(Contract_ID)ON UPDATE CASCADE);
